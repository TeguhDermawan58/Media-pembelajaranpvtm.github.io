    <script>
        // --- LOGIKA MATRIX RAIN ---
        const canvas = document.getElementById('matrixCanvas');
        const ctx = canvas.getContext('2d');

        // Set ukuran canvas full screen
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;

        const katakana = 'アァカサタナハマヤャラワガザダバパイィキシチニヒミリヂビピウゥクスツヌフムユュルグズブヅプエェケセテネヘメレゲゼデベペオォコソトノホモヨョロヲゴゾドボポヴッン';
        const latin = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
        const nums = '0123456789';
        const alphabet = katakana + latin + nums;

        const fontSize = 16;
        const columns = canvas.width / fontSize;
        const drops = [];

        // Inisialisasi posisi drop (semua mulai dari atas - y=1)
        for (let x = 0; x < columns; x++) {
            drops[x] = 1;
        }

        function drawMatrix() {
            // Background hitam transparan untuk efek jejak (trail)
            ctx.fillStyle = 'rgba(0, 0, 0, 0.05)';
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            ctx.fillStyle = '#0F0'; // Warna Hijau Matrix
            ctx.font = fontSize + 'px monospace';

            for (let i = 0; i < drops.length; i++) {
                const text = alphabet.charAt(Math.floor(Math.random() * alphabet.length));
                ctx.fillText(text, i * fontSize, drops[i] * fontSize);

                if (drops[i] * fontSize > canvas.height && Math.random() > 0.975) {
                    drops[i] = 0;
                }
                drops[i]++;
            }
        }

        // Jalankan animasi matrix setiap 30ms
        const matrixInterval = setInterval(drawMatrix, 30);

        // Handle resize window agar matrix tidak pecah
        window.addEventListener('resize', () => {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        });


        // --- LOGIKA UTAMA APLIKASI ---
        document.addEventListener('DOMContentLoaded', function() {
            
            // --- CONFIG ---
            const imagePath = "Pendingin/"; 
            const totalFiles = 25; 
            
            // Variabel untuk Preloading
            let imagesLoadedCount = 0;
            const loadingText = document.getElementById('loadingText');
            const loaderContainer = document.getElementById('loader-container');

            // Fungsi update progress bar
            function updateProgress() {
                imagesLoadedCount++;
                const percentage = Math.floor((imagesLoadedCount / totalFiles) * 100);
                loadingText.innerText = `MENGUNDUH DATA SISTEM... ${percentage}%`;

                // Jika semua gambar sudah selesai dimuat
                if (imagesLoadedCount >= totalFiles) {
                    setTimeout(() => {
                        // Hentikan animasi matrix & hilangkan loader
                        clearInterval(matrixInterval);
                        loaderContainer.style.opacity = '0';
                        setTimeout(() => {
                            loaderContainer.style.display = 'none';
                        }, 1000); // Tunggu efek fade out selesai
                    }, 500); // Delay sedikit agar user melihat 100%
                }
            }

            // --- DATA GENERATOR ---
            const pagesData = [];

            for (let i = 1; i <= totalFiles; i++) {
                let pageInfo = {
                    index: i - 1,
                    file: `mesin (${i}).png`,
                    hotspots: []
                };

                // --- Menambahkan Fungsi Navigasi (Sama seperti sebelumnya) ---
                if (i > 1) { 
                    pageInfo.hotspots.push({ top: 12, left: 8, width: 13, height: 10, action: 'flip', target: 0, label: 'Ke Cover' });
                    pageInfo.hotspots.push({ top: 90, left: 5, width: 15, height: 8, action: 'prev', label: 'Halaman Sebelumnya' });
                    if (i < totalFiles) {
                        pageInfo.hotspots.push({ top: 90, left: 80, width: 15, height: 8, action: 'next', label: 'Halaman Selanjutnya' });
                    }
                }

                // --- Hotspot Daftar Isi ---
                if (i === 2) {
                    pageInfo.hotspots.push({ top: 33, left: 10, width: 20, height: 15, action: 'flip', target: 4 });
                    pageInfo.hotspots.push({ top: 33, left: 40, width: 20, height: 15, action: 'flip', target: 5 });
                    pageInfo.hotspots.push({ top: 33, left: 70, width: 20, height: 15, action: 'flip', target: 3 });
                    pageInfo.hotspots.push({ top: 55, left: 10, width: 20, height: 15, action: 'flip', target: 8 });
                    pageInfo.hotspots.push({ top: 55, left: 40, width: 20, height: 15, action: 'flip', target: 20 });
                    pageInfo.hotspots.push({ top: 55, left: 70, width: 20, height: 15, action: 'flip', target: 24 });
                }

                pagesData.push(pageInfo);
            }

            const flipbookEl = document.getElementById('flipbook');
            const tocListEl = document.getElementById('tocList');

            // --- RENDER HALAMAN DENGAN PRELOADER ---
            pagesData.forEach((page, index) => {
                const pageDiv = document.createElement('div');
                pageDiv.classList.add('page');
                if(index === 0 || index === pagesData.length - 1) pageDiv.setAttribute('data-density', 'hard');

                // PENTING: Menambahkan event 'onload' dan 'onerror' pada gambar
                // agar progress bar berjalan sesuai status download gambar
                let imgElement = `<img src="${imagePath}${page.file}" class="page-image" draggable="false" 
                                  onload="imageLoadedSuccess()" onerror="imageLoadedSuccess()">`;
                
                let hotspotsHtml = '';
                page.hotspots.forEach(hs => {
                    let clickFn = '';
                    if(hs.action === 'flip') clickFn = `flipToPage(${hs.target})`;
                    else if(hs.action === 'prev') clickFn = `pageFlip.flipPrev()`;
                    else if(hs.action === 'next') clickFn = `pageFlip.flipNext()`;

                    hotspotsHtml += `<div class="hotspot" 
                        style="top:${hs.top}%; left:${hs.left}%; width:${hs.width}%; height:${hs.height}%;"
                        onclick="${clickFn}" title="${hs.label || ''}"></div>`;
                });

                pageDiv.innerHTML = `<div class="page-content">${imgElement}<div class="hotspot-layer">${hotspotsHtml}</div></div>`;
                flipbookEl.appendChild(pageDiv);

                // --- Sidebar Logic ---
                const titles = {
                    0: "Cover Depan", 1: "Daftar Isi", 2: "Kata Pengantar", 3: "Petunjuk Penggunaan",
                    4: "Peta Konsep", 5: "Uraian Materi", 6: "Tujuan Pembelajaran", 7: "Prinsip Kerja",
                    8: "Pendingin Udara", 9: "Pendingin Cairan 1", 10: "Pendingin Cairan 2",
                    13: "Radiator", 14: "Tutup Radiator", 15: "Water Pump", 22: "Diagnosa Overheat", 24: "Daftar Pustaka"
                };
                
                if (titles[index] || index % 5 === 0) {
                    const titleText = titles[index] || `Halaman ${index + 1}`;
                    const tocItem = document.createElement('div');
                    tocItem.className = 'toc-item';
                    tocItem.innerHTML = `<span>${titleText}</span> <span style="font-size:11px; opacity:0.5">${index + 1}</span>`;
                    tocItem.onclick = () => { flipToPage(index); toggleSidebar(false); };
                    tocListEl.appendChild(tocItem);
                }
            });

            // Fungsi global agar bisa dipanggil dari atribut onload string HTML
            window.imageLoadedSuccess = function() {
                updateProgress();
            };

            // --- INISIALISASI PAGEFLIP ---
            const pageFlip = new St.PageFlip(flipbookEl, {
                width: 450, height: 637,
                size: "stretch", minWidth: 300, maxWidth: 1000,
                minHeight: 400, maxHeight: 1200,
                maxShadowOpacity: 0.5, showCover: true, mobileScrollSupport: false
            });

            pageFlip.loadFromHTML(document.querySelectorAll('.page'));

            // --- FUNGSI GLOBAL & EVENTS (Sama seperti sebelumnya) ---
            window.pageFlip = pageFlip;
            window.flipToPage = (idx) => pageFlip.flip(idx);

            document.getElementById('extPrevBtn').onclick = () => pageFlip.flipPrev();
            document.getElementById('extNextBtn').onclick = () => pageFlip.flipNext();

            const sidebar = document.getElementById('sidebar');
            function toggleSidebar(show) {
                if (show) sidebar.classList.add('show');
                else sidebar.classList.remove('show');
            }
            document.getElementById('btnToggleSidebar').onclick = () => toggleSidebar(!sidebar.classList.contains('show'));
            document.getElementById('btnCloseSidebar').onclick = () => toggleSidebar(false);

            let currentZoom = 1;
            const bookDiv = document.getElementById('flipbook');
            document.getElementById('btnZoomIn').onclick = () => {
                if (currentZoom < 1.5) { currentZoom += 0.1; bookDiv.style.transform = `scale(${currentZoom})`; }
            };
            document.getElementById('btnZoomOut').onclick = () => {
                if (currentZoom > 0.6) { currentZoom -= 0.1; bookDiv.style.transform = `scale(${currentZoom})`; }
            };
            document.getElementById('btnFullscreen').onclick = () => {
                if (!document.fullscreenElement) document.documentElement.requestFullscreen();
                else if (document.exitFullscreen) document.exitFullscreen();
            };

            document.addEventListener('keydown', (e) => {
                if (e.key === "ArrowRight") pageFlip.flipNext();
                if (e.key === "ArrowLeft") pageFlip.flipPrev();
            });
        });
    </script>
